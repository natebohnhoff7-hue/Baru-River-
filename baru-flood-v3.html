-- Automated readings (logged each app load) CREATE TABLE flood_readings ( id BIGSERIAL PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), score NUMERIC(4,3), risk_level TEXT, rain_24h NUMERIC(6,1), rain_72h NUMERIC(6,1), soil NUMERIC(5,3), tide_m NUMERIC(4,2), lunar_phase TEXT, ws_avg_24h NUMERIC(6,1), ws_avg_72h NUMERIC(6,1) ); -- Manual field observations CREATE TABLE observations ( id BIGSERIAL PRIMARY KEY, created_at TIMESTAMPTZ DEFAULT NOW(), condition TEXT, rainfall TEXT, water_color TEXT, notes TEXT, observer TEXT, flood_score NUMERIC(4,3) ); -- Allow anonymous reads/writes (Row Level Security) ALTER TABLE flood_readings ENABLE ROW LEVEL SECURITY; ALTER TABLE observations ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_read" ON flood_readings FOR SELECT USING (true); CREATE POLICY "public_write" ON flood_readings FOR INSERT WITH CHECK (true); CREATE POLICY "public_read" ON observations FOR SELECT USING (true); CREATE POLICY "public_write" ON observations FOR INSERT WITH CHECK (true);
